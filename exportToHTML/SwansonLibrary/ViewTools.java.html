103      static public Line getPerpindicularCenterIntersectedLine(Line originalLine){ 
104   
105   
106          int scaleFactor = 4; 
107   
108          int dx = originalLine.end.x - originalLine.start.x; 
109          int dy = originalLine.end.y - originalLine.start.y; 
110   
111          int orginalLineMidX = originalLine.start.x + Math.round(.5f * dx); 
112          int orginalLineMidY = originalLine.start.y + Math.round(.5f * dy); 
113   
114   
115          int perpdx = dy; 
116          int perpdy = dx; 
117   
118          int perpLineStartX = orginalLineMidX + ((scaleFactor * perpdx)*-1); 
119          int perpLineStartY = orginalLineMidY + (scaleFactor * perpdy); 
120          int perpLineEndX = orginalLineMidX + (scaleFactor * perpdx); 
121          int perpLineEndY = orginalLineMidY + ((scaleFactor * perpdy)*-1); 
122   
123          //return new Point(orginalLineMidX,orginalLineMidY); 
124          return new ViewTools.Line(perpLineStartX,perpLineStartY,perpLineEndX,perpLineEndY); 
125      }

127      // Returns 1 if the lines intersect, otherwise 0. In addition, if the lines 
128      // intersect the intersection point may be stored in the floats i_x and i_y. 
129      static public Intersection get_line_intersection(Line lineOne, Line lineTwo){ 
130   
131          Intersection thisIntersection = new Intersection(); 
132   
133   
134          float s1_x, s1_y, s2_x, s2_y; 
135          s1_x = lineOne.end.x - lineOne.start.x;     s1_y = lineOne.end.y - lineOne.start.y; 
136          s2_x = lineTwo.end.x - lineTwo.start.x;     s2_y = lineTwo.end.y - lineTwo.start.y; 
137   
138          float s, t; 
139          s = (-s1_y * (lineOne.start.x - lineTwo.start.x) + s1_x * (lineOne.start.y - lineTwo.start.y)) / (-s2_x * s1_y + s1_x * s2_y); 
140          t = ( s2_x * (lineOne.start.y - lineTwo.start.y) - s2_y * (lineOne.start.x - lineTwo.start.x)) / (-s2_x * s1_y + s1_x * s2_y); 
141   
142          if (s >= 0 && s <= 1 && t >= 0 && t <= 1) 
143          { 
144              // Collision detected 
145              thisIntersection.intersects=true; 
146              thisIntersection.intersectionPoint.x = (int) (lineOne.start.x + (t * s1_x)); 
147              thisIntersection.intersectionPoint.y = (int) (lineOne.start.y +  (t * s1_y)); 
148   
149          } 
150   
151          return thisIntersection; 
152      }

154      static public Point vectorToPoint(double vectorinRadians, double length, Point start, Rect bounds){ 
155          Point point = vectorToPoint(vectorinRadians, length, start); 
156   
157          if(!containsInner(point, bounds)){ 
158              if(point.x<=bounds.left) point.x=bounds.left+1; 
159              if(point.x>=bounds.right) point.x=bounds.right-1; 
160              if(point.y>=bounds.top) point.y=bounds.top+1; 
161              if(point.y<=bounds.bottom) point.y=bounds.bottom-1; 
162          } 
163   
164   
165          return point; 
166      }

207      static public double getAngleDotProduct(Point p1, Point p2, Point p3){ 
208          float x1 = p1.x; 
209          float x2 = p2.x; 
210          float x3 = p3.x; 
211          float x4 = p2.x; 
212   
213          float y1 = p1.y; 
214          float y2 = p2.y; 
215          float y3 = p3.y; 
216          float y4 = p2.y; 
217   
218          float dx1 = x2-x1; 
219          float dy1 = y2-y1; 
220          float dx2 = x4-x3; 
221          float dy2 = y4-y3; 
222   
223          float d = dx1*dx2 + dy1*dy2;   // dot product of the 2 vectors 
224          float l2 = (dx1*dx1+dy1*dy1)*(dx2*dx2+dy2*dy2); // product of the squared lengths 
225   
226          return Math.toDegrees(Math.acos(d / Math.sqrt(l2))); 
227      }

266      public static Rect squareRect(Rect rect){ 
267          Rect smallerRect = new Rect(rect); 
268          if(rect.width()==rect.height())return smallerRect; 
269          else if(rect.width()<rect.height()){ 
270              smallerRect.top=rect.centerY()-rect.width()/2; 
271              smallerRect.bottom=rect.centerY()+rect.width()/2; 
272          }else{ 
273              smallerRect.left=rect.centerX()+rect.height()/2; 
274              smallerRect.right=rect.centerX()-rect.height()/2; 
275          } 
276          return smallerRect; 
277      }
