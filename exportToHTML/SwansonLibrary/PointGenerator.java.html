69       public class RadianExclusionMap{ 
70           private ArrayList<RadianRange> mExcludedRanges = new ArrayList<RadianRange>(); 
71           private Point mCurrentPoint; 
72           private double mAvailableRadians = ViewTools.FULLCIRCLE; 
73           private double mTotalExludedRadians = 0; 
74    
75           private RadianExclusionMap(Point Origin, Point CurrentPoint) { 
76    
77               mCurrentPoint=CurrentPoint; 
78    
79               //opposite angle exclusion 
80               Double lastRadian = ViewTools.getArcTan2Mapped(Origin,CurrentPoint); 
81               double beginingRad; double endRad; 
82               beginingRad = (lastRadian + ViewTools.FULLCIRCLE - (mMinAngle*Math.PI) + Math.PI)%ViewTools.FULLCIRCLE; 
83               endRad = (lastRadian + (mMinAngle*Math.PI) + Math.PI)%ViewTools.FULLCIRCLE; 
84    
85    
86               if(beginingRad>endRad){ 
87                   mExcludedRanges.add(new RadianRange(beginingRad,ViewTools.FULLCIRCLE)); 
88                   beginingRad=0; 
89               } 
90               mExcludedRanges.add(new RadianRange(beginingRad,endRad)); 
91    
92    
93               //detect proximity to walls, exclude them 
94    
95               if(mCurrentPoint.y-mBounds.top<mMinLength){ 
96    
97                   mExcludedRanges.add(getWallRadianRange(mBounds.top, ViewTools.RadianAngles.TOP, mCurrentPoint.y, true)); 
98               } 
99    
100              if(mBounds.bottom-mCurrentPoint.y<mMinLength){ 
101                  mExcludedRanges.add(getWallRadianRange(mBounds.bottom, ViewTools.RadianAngles.BOTTOM,mCurrentPoint.y,true)); 
102              } 
103              if(mCurrentPoint.x-mBounds.left<mMinLength){ 
104                  mExcludedRanges.add(getWallRadianRange(mBounds.left, ViewTools.RadianAngles.LEFT,mCurrentPoint.x,false)); 
105              } 
106              if(mBounds.right-mCurrentPoint.x<mMinLength){ 
107                  mExcludedRanges.add(getWallRadianRange(mBounds.right, ViewTools.RadianAngles.RIGHT,mCurrentPoint.x,false)); 
108              } 
109   
110              //sort and merge 
111              if(mExcludedRanges.size()>1){ 
112                  mExcludedRanges=SortMergeImroved(); 
113              } 
114   
115              //calc available 
116              for(int i=0;i<mExcludedRanges.size();i++){ 
117                  mTotalExludedRadians+=mExcludedRanges.get(i).range(); 
118                  mAvailableRadians-=mExcludedRanges.get(i).range(); 
119              } 
120   
121   
122   
123          }

127          private ArrayList<RadianRange> SortMergeImroved(){ 
128              ArrayList<RadianRange> unSortedRanges = (ArrayList<RadianRange>) mExcludedRanges.clone(); 
129              ArrayList<RadianRange> sortedRanges = new ArrayList<RadianRange>(); 
130              ArrayList<RadianRange> mergedRanges = new ArrayList<RadianRange>(); 
131   
132   
133              //sort 
134              while (unSortedRanges.size()>0){ 
135   
136                  RadianRange smallestRadian=unSortedRanges.get(0); 
137                  for(int v=0;v<unSortedRanges.size();v++){ 
138                      if(unSortedRanges.get(v).begining<smallestRadian.begining) { 
139                          smallestRadian=unSortedRanges.get(v); 
140                      } 
141                  } 
142                  sortedRanges.add(smallestRadian); 
143                  unSortedRanges.remove(smallestRadian); 
144   
145              } 
146   
147              //merge 
148              for(int i=0;i<sortedRanges.size()-1;i++){ 
149                  if(sortedRanges.get(i).end>sortedRanges.get(i+1).begining){ 
150                      sortedRanges.get(i+1).begining=sortedRanges.get(i).begining; 
151                  }else mergedRanges.add(sortedRanges.get(i)); 
152              } 
153              mergedRanges.add(sortedRanges.get(sortedRanges.size()-1)); 
154   
155              return (ArrayList<RadianRange>)mergedRanges.clone(); 
156   
157          }

159          private RadianRange getWallRadianRange(int walledge, double wallRadian, int point, boolean y) { 
160   
161   
162              int opposite = point-walledge; 
163              Double adjecent = Math.sqrt((mMinLength*mMinLength-opposite*opposite)); 
164              double rad; 
165              if(y)rad=ViewTools.getArcTan2Mapped(opposite, adjecent); 
166              else rad=ViewTools.getArcTan2Mapped(adjecent, opposite); 
167   
168              double wallRadStart = (rad+Math.PI)%ViewTools.FULLCIRCLE;; 
169              double wallRadEnd = (wallRadian + (wallRadian-wallRadStart)+ ViewTools.FULLCIRCLE)%ViewTools.FULLCIRCLE; 
170   
171   
172   
173              if(wallRadEnd<wallRadStart && wallRadian!=0){ 
174                  double radtemp=wallRadStart; 
175                  wallRadStart=wallRadEnd; 
176                  wallRadEnd=radtemp; 
177              } 
178   
179              if(wallRadian==0){ 
180                  //mAvailableRadians-=(ViewTools.FULLCIRCLE-wallRadStart); 
181   
182                  mExcludedRanges.add(new RadianRange(wallRadStart,ViewTools.FULLCIRCLE)); //find a much better way for this! 
183   
184                  wallRadStart=0; 
185              } 
186   
187   
188   
189              return new RadianRange(wallRadStart,wallRadEnd); 
190   
191          }

232          private double getAppropriateRad() { 
233              double unmappedRad = mAvailableRadians*randomGen.nextFloat(); 
234   
235              return getMappedRad(unmappedRad); 
236          } 
237          private double getMappedRad(double unmappedRad){ 
238   
239              for(int i=0;i<mExcludedRanges.size();i++){ 
240                  if(unmappedRad>mExcludedRanges.get(i).begining){ 
241                      unmappedRad+=mExcludedRanges.get(i).range(); 
242                  } 
243              } 
244   
245              return unmappedRad; 
246          }

249          private double getAppropriateLenth(double approvedRad) { 
250              //get point off frame 
251              Point offScreenPoint=ViewTools.vectorToPoint(approvedRad, mBoundsDiagonal, mCurrentPoint);
252              //find intersecton 
253   
254              ViewTools.Line extendedLine = new ViewTools.Line(mCurrentPoint.x,mCurrentPoint.y,offScreenPoint.x,offScreenPoint.y); 
255   
256              ViewTools.Intersection intersections[]= new ViewTools.Intersection[4]; 
257   
258              ArrayList<Point> borderpoints = new ArrayList<Point>(); 
259              Point borderpoint = new Point(-1,-1); 
260   
261              int numIntersections=0; 
262   
263              for(int i=0;i<4;i++){ 
264                  intersections[i]=ViewTools.get_line_intersection(extendedLine,BorderLines[i]); 
265                  if(intersections[i].intersects) { 
266                      numIntersections++; 
267                      borderpoints.add(intersections[i].intersectionPoint); 
268                  } 
269              } 
270              if(numIntersections>1){ 
271                 double greatestDistance=0; 
272                 for(int i=0;i<borderpoints.size();i++){ 
273                     if(ViewTools.getDistancetoNonSQRTD(mCurrentPoint,borderpoints.get(i))>greatestDistance){ 
274                         borderpoint=borderpoints.get(i); 
275                     } 
276                 } 
277              } 
278              else if(numIntersections==0){ 
279                  Log.d("PATH", "border intersection not detected"); 
280              } 
281              else borderpoint = borderpoints.get(0); 
282   
283   
284   
285              //use to calc distance 
286              Double WallLength=Math.sqrt(ViewTools.getDistancetoNonSQRTD(mCurrentPoint.x, mCurrentPoint.y, borderpoint.x, borderpoint.y)); 
287   
288   
289              //get random length 
290   
291   
292              double pointLength; 
293   
294              if(WallLength>=mMinLength){ 
295                  pointLength=randomGen.nextFloat()*(WallLength-mMinLength)+mMinLength; //for rounding error 
296              }else{ 
297   
298                  pointLength=WallLength; 
299   
300              } 
301              if(pointLength==0){ 
302                  Log.d("PATH", "zero length"); 
303              } 
304              return pointLength; 
305          }
