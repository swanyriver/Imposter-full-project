62       public static void getDrawableSegment(Path segment, Path dstPath, float granularity){ 
63           dstPath.rewind(); 
64           PathMeasure pathMeasure = new PathMeasure(segment,false); 
65           float length = pathMeasure.getLength(); 
66           ViewTools.PathPosition pathPosition = ViewTools.getPathPosition(pathMeasure,0); 
67           dstPath.moveTo(pathPosition.position.x,pathPosition.position.y); 
68           for(float distance=0;distance<=length;distance+=granularity){ 
69               pathPosition=ViewTools.getPathPosition(pathMeasure,distance); 
70               dstPath.lineTo(pathPosition.position.x,pathPosition.position.y); 
71           } 
72    
73       }
